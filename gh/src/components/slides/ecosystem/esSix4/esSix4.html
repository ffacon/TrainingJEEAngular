Ecosystem ES6 Introduction - Setting default parameters values

Javascript in general is a very permissive language, and allows to do stuff like:

  • define functions with parameters, but call it without these parameters
  • Oppositely, not define any parameters in your function but to access them if the function is called with parameters (with the arguments keyword)

ES6 now allow to define default values for parameters

//For information only function addPonies() { for (var i = 0; i < arguments.length; i++) { console.log(arguments[i]); } } addPonies('Rainbow Dash', 'Pinkie Pie'); //The old way var defaultParamValues1= function(param1, param2){ var param1Val= param1 || 'defaultVal1'; var param2Val= param2 || 'defaultVal2'; console.log(param1Val + ', ' + param2Val); }; defaultParamValues1(); // defaultVal1 , defaultVal2 defaultParamValues1('custom val1', 'custom val2'); // custom val1 , custom val2 //The new way let defaultParamValues2= function(param1 = 'defaultVal1', param2= 'defaultVal2'){ console.log(param1 + ', ' + param2); }; defaultParamValues2(); // defaultVal1 , defaultVal2 defaultParamValues2('custom val1', 'custom val2'); // custom val1 , custom val2 //The old way function drawES5Chart(options) { options = options === undefined ? {} : options; var size = options.size === undefined ? 'big' : options.size; var cords = options.cords === undefined ? { x: 0, y: 0 } : options.cords; var radius = options.radius === undefined ? 25 : options.radius; console.log(size, cords, radius); // now finally do some chart drawing } drawES5Chart({ cords: { x: 18, y: 30 }, radius: 30 }); function drawES6Chart({size = 'big', cords = { x: 0, y: 0 }, radius = 25} = {}) { console.log(size, cords, radius); // do some chart drawing } drawES6Chart({ cords: { x: 18, y: 30 }, radius: 30 });
