Main Concepts Router - Basics

  • Below is the most basic example
  • First we configure our routes through the RouterModule.forRoot method, this allow us to retrieve a new configured module
  • Once routes are configured, we need:
    • to define the zone where the view will be displayed: the router-outlet directive
    • A way to navigate from a view to another: the routerLink directive

Below is a first example of routing defined at the main application component level

import {Component} from '@angular/core'; import {BrowserModule} from '@angular/platform-browser'; import {Home} from './home.component'; import {TodoList} from './todolist.component'; import {TodoDetails} from './tododetails.component'; import {routesModule} from './app.routing'; @Component({ selector:'app', template:` <h1> min app </h1> <ul> <li> <a [routerLink]="['home']">home</a> </li> <li> <a [routerLink]="['todolist']">todo list</a> </li> <li> <a [routerLink]="['tododetails']">todo details</a> </li> </ul> <router-outlet></router-outlet> ` }) class AppComponent {} export const ModuleDefs= { imports:[BrowserModule, routesModule], declarations:[AppComponent, Home, TodoList, TodoDetails], bootstrap: [AppComponent] } import {RouterModule} from '@angular/router'; import {Home} from './home.component'; import {TodoList} from './todolist.component'; import {TodoDetails} from './tododetails.component'; const routesDefs= [ { path:'home', component: Home }, { path: 'todolist', component: TodoList }, { path: 'tododetails', component: TodoDetails }, { path: '', redirectTo: 'home', pathMatch: 'full'} ]; export const routesModule= RouterModule.forRoot(routesDefs); import {Component} from '@angular/core'; @Component({ selector:'home', template:'Home' }) export class Home{} import {Component} from '@angular/core'; @Component({ selector:'todolist', template:'Todo List' }) export class TodoList{} import {Component} from '@angular/core'; @Component({ selector:'tododetails', template:'Todo details' }) export class TodoDetails{}
