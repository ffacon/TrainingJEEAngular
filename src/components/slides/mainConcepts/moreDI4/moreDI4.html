Main Concepts Dependency Injection - More Details - Provide

And we can even do more with provide

  • configure simple values (singletons)
  • configure aliases (singletons)
  • configure factories
import {ReflectiveInjector} from '@angular/core'; //Define our injector and its providers let injector = ReflectiveInjector.resolveAndCreate([ {provide: 'AppVersion', useValue: '1.0.0-SNAPSHOT'} ]); import {ReflectiveInjector} from '@angular/core'; //Define our injector and its providers let injector = ReflectiveInjector.resolveAndCreate([ { provide: Person, useClass: PersonImpl} ]); import {ReflectiveInjector} from '@angular/core'; //Define our injector and its providers let injector = ReflectiveInjector.resolveAndCreate([ { provide: PersonFactory, useFactory: (loggerService) => { return new RandomPerson(loggerService); }, deps: [LoggerService] } ]);
